// iostream, stack and vector are included

vector<int> nextGreater(vector<int>& arr){
    int n=arr.size();
    vector<int> res(n);
    stack<int> st;
    for(int i=n-1;i>=0;i--){
        while(!st.empty() && st.top() <=arr[i]){
            st.pop();
        }
        if(st.empty()){
            res[i]=-1;
        }
        else{
            res[i]=st.top();
        }
        st.push(arr[i]);
    }
    return res;
    
}
